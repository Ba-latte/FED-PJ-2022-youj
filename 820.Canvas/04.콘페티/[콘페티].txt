export function hexToRgb(hex){  // #FF0000 or #ff0000
    var result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
    return result ? {

        b: parseInt(result[3], 16)  // 00 -> 0
    } : null;
